Design Patterns

Singleton Pattern

public class Singleton {
    private static Singleton instance;
    private Singleton() {}
    
    public static synchronized Singleton getInstance() {
        if (instance == null) {
            instance = new Singleton();
        }
        return instance;
    }
}

Factory Pattern

interface Animal {
    void makeSound();
}

class Dog implements Animal {
    public void makeSound() { System.out.println("Woof!"); }
}

class Cat implements Animal {
    public void makeSound() { System.out.println("Meow!"); }
}

class AnimalFactory {
    public static Animal createAnimal(String type) {
        switch(type.toLowerCase()) {
            case "dog": return new Dog();
            case "cat": return new Cat();
            default: throw new IllegalArgumentException("Unknown animal type");
        }
    }
}

Observer Pattern

interface Observer {
    void update(String message);
}

class Subject {
    private List observers = new ArrayList<>();
    
    public void attach(Observer observer) {
        observers.add(observer);
    }
    
    public void notifyObservers() {
        for (Observer observer : observers) {
            observer.update("New update!");
        }
    }
}

Object-Oriented Design Principles (SOLID)

Single Responsibility Principle (SRP)

A class should have only one reason to change, meaning it should have only one job or responsibility.

// Good example
class UserService {
    public void createUser() { /* ... */ }
    public void deleteUser() { /* ... */ }
}

class EmailService {
    public void sendEmail() { /* ... */ }
    public void validateEmail() { /* ... */ }
}

Open/Closed Principle (OCP)

Software entities should be open for extension but closed for modification.

abstract class Shape {
    abstract double area();
}

class Circle extends Shape {
    private double radius;
    public double area() { return Math.PI * radius * radius; }
}

class Rectangle extends Shape {
    private double width, height;
    public double area() { return width * height; }
}

Liskov Substitution Principle (LSP)

Objects of a superclass should be replaceable with objects of its subclasses without breaking the application.

class Bird {
    public void fly() { /* ... */ }
}

class Sparrow extends Bird {
    public void fly() { /* ... */ }
}

class Penguin extends Bird {
    public void fly() {
        throw new UnsupportedOperationException("Penguins can't fly!");
    }
}

Interface Segregation Principle (ISP)

A client should not be forced to implement interfaces they don't use.

interface Workable {
    void work();
}

interface Eatable {
    void eat();
}

class Human implements Workable, Eatable {
    public void work() { /* ... */ }
    public void eat() { /* ... */ }
}

class Robot implements Workable {
    public void work() { /* ... */ }
}

Dependency Inversion Principle (DIP)

High-level modules should not depend on low-level modules. Both should depend on abstractions.

interface Database {
    void connect();
    void query();
}

class MySQLDatabase implements Database {
    public void connect() { /* ... */ }
    public void query() { /* ... */ }
}

class UserRepository {
    private Database database;
    public UserRepository(Database database) {
        this.database = database;
    }
}

Design a Hash Map

class HashMap {
    private static class Node {
        K key;
        V value;
        Node next;
        Node(K key, V value) { this.key = key; this.value = value; }
    }
    private Node[] buckets;
    private int capacity = 16;
    
    public HashMap() { buckets = new Node[capacity]; }
    
    public void put(K key, V value) {
        int index = key.hashCode() % capacity;
        Node node = new Node<>(key, value);
        node.next = buckets[index];
        buckets[index] = node;
    }
    
    public V get(K key) {
        int index = key.hashCode() % capacity;
        Node current = buckets[index];
        while (current != null) {
            if (current.key.equals(key)) return current.value;
            current = current.next;
        }
        return null;
    }
}

Design a Least Recently Used (LRU) Cache

import java.util.*;

class LRUCache {
    private final int capacity;
    private final Map cache;
    
    public LRUCache(int capacity) {
        this.capacity = capacity;
        this.cache = new LinkedHashMap<>(capacity, 0.75f, true) {
            protected boolean removeEldestEntry(Map.Entry eldest) {
                return size() > capacity;
            }
        };
    }
    
    public V get(K key) { return cache.getOrDefault(key, null); }
    public void put(K key, V value) { cache.put(key, value); }
}

Design a Call Center

class Call {
    private String id;
    private String customer;
    private long timestamp;
    private CallStatus status;
    
    public Call(String id, String customer) {
        this.id = id;
        this.customer = customer;
        this.timestamp = System.currentTimeMillis();
        this.status = CallStatus.WAITING;
    }
}

enum CallStatus {
    WAITING, IN_PROGRESS, COMPLETED
}

class CallCenter {
    private Queue callQueue = new LinkedList<>();
    private List agents = new ArrayList<>();
    
    public void receiveCall(Call call) {
        callQueue.offer(call);
        dispatchCall();
    }
    
    private void dispatchCall() {
        Agent availableAgent = findAvailableAgent();
        if (availableAgent != null && !callQueue.isEmpty()) {
            Call call = callQueue.poll();
            availableAgent.handleCall(call);
        }
    }
    
    private Agent findAvailableAgent() {
        return agents.stream()
                    .filter(Agent::isAvailable)
                    .findFirst()
                    .orElse(null);
    }
}

class Agent {
    private String id;
    private boolean available = true;
    
    public void handleCall(Call call) {
        available = false;
        // Handle call logic
        available = true;
    }
    
    public boolean isAvailable() { return available; }
}

Design a Deck of Cards

enum Suit { HEARTS, DIAMONDS, CLUBS, SPADES; }

class Card {
    private int value;
    private Suit suit;
    
    public Card(int value, Suit suit) {
        this.value = value;
        this.suit = suit;
    }
    
    public int getValue() { return value; }
    public Suit getSuit() { return suit; }
    
    @Override
    public String toString() {
        return value + " of " + suit;
    }
}

class Deck {
    private List cards = new ArrayList<>();
    
    public Deck() {
        initializeDeck();
        shuffle();
    }
    
    private void initializeDeck() {
        for (Suit suit : Suit.values()) {
            for (int i = 1; i <= 13; i++) {
                cards.add(new Card(i, suit));
            }
        }
    }
    
    public void shuffle() {
        Collections.shuffle(cards);
    }
    
    public Card drawCard() {
        if (cards.isEmpty()) return null;
        return cards.remove(cards.size() - 1);
    }
    
    public void reset() {
        cards.clear();
        initializeDeck();
        shuffle();
    }
}

Design a Parking Lot

enum VehicleType {
    MOTORCYCLE, CAR, BUS
}

class Vehicle {
    private String licensePlate;
    private VehicleType type;
    
    public Vehicle(String licensePlate, VehicleType type) {
        this.licensePlate = licensePlate;
        this.type = type;
    }
}

class ParkingSpot {
    private int spotNumber;
    private VehicleType type;
    private boolean isOccupied;
    private Vehicle parkedVehicle;
    
    public ParkingSpot(int spotNumber, VehicleType type) {
        this.spotNumber = spotNumber;
        this.type = type;
        this.isOccupied = false;
    }
    
    public boolean canPark(Vehicle vehicle) {
        return !isOccupied && vehicle.getType() == type;
    }
    
    public void park(Vehicle vehicle) {
        parkedVehicle = vehicle;
        isOccupied = true;
    }
    
    public void removeVehicle() {
        parkedVehicle = null;
        isOccupied = false;
    }
}

class ParkingLot {
    private Map spots = new HashMap<>();
    private int capacity;
    
    public ParkingLot(int capacity) {
        this.capacity = capacity;
        initializeSpots();
    }
    
    private void initializeSpots() {
        // Initialize different types of spots
        for (int i = 1; i <= capacity; i++) {
            VehicleType type = determineSpotType(i);
            spots.put(i, new ParkingSpot(i, type));
        }
    }
    
    public boolean parkVehicle(Vehicle vehicle) {
        for (ParkingSpot spot : spots.values()) {
            if (spot.canPark(vehicle)) {
                spot.park(vehicle);
                return true;
            }
        }
        return false;
    }
    
    public void removeVehicle(String licensePlate) {
        for (ParkingSpot spot : spots.values()) {
            if (spot.isOccupied() && 
                spot.getParkedVehicle().getLicensePlate().equals(licensePlate)) {
                spot.removeVehicle();
                break;
            }
        }
    }
}

Design a Chat Server

class Message {
    private String sender;
    private String content;
    private long timestamp;
    
    public Message(String sender, String content) {
        this.sender = sender;
        this.content = content;
        this.timestamp = System.currentTimeMillis();
    }
}

class ChatRoom {
    private String name;
    private List users = new ArrayList<>();
    private List messages = new ArrayList<>();
    
    public void addUser(User user) {
        users.add(user);
    }
    
    public void removeUser(User user) {
        users.remove(user);
    }
    
    public void broadcast(Message message) {
        messages.add(message);
        for (User user : users) {
            user.receiveMessage(message);
        }
    }
}

class User {
    private String username;
    private List rooms = new ArrayList<>();
    
    public void joinRoom(ChatRoom room) {
        rooms.add(room);
        room.addUser(this);
    }
    
    public void leaveRoom(ChatRoom room) {
        rooms.remove(room);
        room.removeUser(this);
    }
    
    public void sendMessage(ChatRoom room, String content) {
        Message message = new Message(username, content);
        room.broadcast(message);
    }
    
    public void receiveMessage(Message message) {
        // Handle received message
    }
}

class ChatServer {
    private Map rooms = new HashMap<>();
    private Map users = new HashMap<>();
    
    public ChatRoom createRoom(String name) {
        ChatRoom room = new ChatRoom(name);
        rooms.put(name, room);
        return room;
    }
    
    public User createUser(String username) {
        User user = new User(username);
        users.put(username, user);
        return user;
    }
}

Design a Circular Array

class CircularArray {
    private T[] array;
    private int head = 0;
    private int size = 0;
    private int capacity;
    
    public CircularArray(int capacity) {
        this.capacity = capacity;
        array = (T[]) new Object[capacity];
    }
    
    public void add(T item) {
        if (size == capacity) {
            // Shift head if array is full
            head = (head + 1) % capacity;
        } else {
            size++;
        }
        array[(head + size - 1) % capacity] = item;
    }
    
    public T get(int index) {
        if (index < 0 || index >= size) {
            throw new IndexOutOfBoundsException();
        }
        return array[(head + index) % capacity];
    }
    
    public void rotate(int shiftRight) {
        head = (head - shiftRight + capacity) % capacity;
    }
    
    public int size() { return size; }
    
    public boolean isEmpty() { return size == 0; }
}
